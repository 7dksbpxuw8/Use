local Players = game:GetService("Players")
local HttpService = game:GetService("HttpService")
local player = Players.LocalPlayer

-- GitHub RAW içeriği: owner=7dksbpxuw8
local content
local success, result = pcall(function()
    return game:HttpGet("https://raw.githubusercontent.com/7dksbpxuw8/Use/main/Auth.txt")
end)

if success then
    content = result
else
    player:Kick("Could not verify owner.")
    return
end

-- Sahip kontrolü
local owner = string.match(content, "owner=(%w+)")
if owner ~= "7dksbpxuw8" then
    player:Kick("Unauthorized GitHub owner.")
    return
end

-- Devam edebilir
print("✅ GitHub sahibi doğrulandı:", owner)
